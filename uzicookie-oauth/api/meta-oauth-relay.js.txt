export default async function handler(req, res) {
  try {
    const { code, state } = req.query;

    if (!code) {
      return res.status(400).json({ error: 'No code provided' });
    }

    // Exchange code for token with Meta
    const tokenResponse = await fetch('https://graph.meta.com/v18.0/oauth/access_token', {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({
        client_id: process.env.META_APP_ID,
        client_secret: process.env.META_APP_SECRET,
        redirect_uri: `${process.env.VERCEL_URL}/api/meta-oauth-relay`,
        code,
      }).toString(),
    });

    const tokenData = await tokenResponse.json();

    if (!tokenResponse.ok || tokenData.error) {
      return res.status(400).json({ error: tokenData.error?.message });
    }

    const accessToken = tokenData.access_token;
    const userId = tokenData.user_id;

    // Get WhatsApp Business Accounts
    const wabaResponse = await fetch(
      `https://graph.meta.com/v18.0/${userId}/whatsapp_business_accounts?fields=id,name,phone_numbers&access_token=${accessToken}`
    );

    const wabaData = await wabaResponse.json();

    if (!wabaResponse.ok || wabaData.error) {
      return res.status(400).json({ error: 'Failed to fetch WhatsApp accounts' });
    }

    const wabaAccounts = wabaData.data || [];
    if (wabaAccounts.length === 0) {
      return res.status(400).json({ error: 'No WhatsApp accounts found' });
    }

    // Get first account and phone
    const waba = wabaAccounts[0];
    const primaryPhone = waba.phone_numbers?.data?.[0];

    if (!primaryPhone) {
      return res.status(400).json({ error: 'No phone numbers found' });
    }

    // Call Base44 backend to store the token
    const storeResponse = await fetch(
      `https://uzicookie.com/api/functions/storeWhatsAppToken`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          accessToken,
          phoneNumber: primaryPhone.phone_number,
          phoneNumberId: primaryPhone.id,
          wabaId: waba.id,
          wabaName: waba.name,
          verifiedName: primaryPhone.verified_name,
        }),
      }
    );

    const storeData = await storeResponse.json();

    if (!storeResponse.ok) {
      return res.status(400).json({ error: storeData.error });
    }

    // Redirect to success page
    return res.redirect(302, 'https://uzicookie.com/SubscriberPortal?wa_connected=true');
  } catch (error) {
    console.error('OAuth relay error:', error);
    return res.status(500).json({ error: error.message });
  }
}
